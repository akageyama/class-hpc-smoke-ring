  subroutine boundary_condition_scalar(scalar)
    !! スカラー場の境界条件設定
    real(DR), dimension(NX,NY,NZ), intent(inout) :: scalar !! スカラー場

    scalar( 1,:,:) = scalar(NX-1,:,:)
    scalar(NX,:,:) = scalar(   2,:,:)

    scalar(:, 1,:) = scalar(:,NY-1,:)
    scalar(:,NY,:) = scalar(:,   2,:)

    scalar(:,:, 1) = scalar(:,:,NZ-1)
    scalar(:,:,NZ) = scalar(:,:,   2)
  end subroutine boundary_condition_scalar
